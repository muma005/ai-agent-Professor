## COMMIT 1A: graph/hackathon_builder.py

### What this does

Builds an alternative LangGraph StateGraph for hackathon mode. Same ProfessorState, same agent functions (imported from the same files), different execution ORDER and different set of nodes. The pipeline inversion: traditional goes data→features→model→writeup. Hackathon goes data→domain→thesis→external data→hypothesis features→model→narrative.

### The graph

```python
"""
Hackathon mode LangGraph graph.

Alternative pipeline for hackathon competitions where the deliverable is
a NARRATIVE (writeup + notebook + visualizations), not just a leaderboard score.

Triggered by: professor run --mode hackathon --competition {name}

Uses the same ProfessorState, same sandbox, same HITL, same Cost Governor.
Different agent ordering, different agents included/excluded.
"""

from langgraph.graph import StateGraph, END
from graph.state import ProfessorState

# === Import shared agents (same code, different orchestration) ===
from shields.preflight import run_preflight_checks
from agents.competition_intel import competition_intel
from agents.data_engineer import data_engineer
from agents.eda_agent import eda_agent
from agents.domain_research import domain_research
from agents.ml_optimizer import ml_optimizer
from agents.red_team_critic import red_team_critic
from agents.feature_factory import feature_factory  # Modified with hackathon mode support
from agents.publisher import publisher as traditional_publisher

# === Import hackathon-specific agents ===
from tools.rubric_parser import run_rubric_parser
from agents.thesis_generator import thesis_generator
from agents.external_data_scout import external_data_scout
from tools.narrative_engine import generate_narrative_plots, generate_hackathon_writeup
from agents.hackathon_publisher import hackathon_publisher

# === Import infrastructure ===
from tools.operator_channel import emit_to_operator, process_pending_injections
from config.constants import MAX_REPLAN_CYCLES


def build_hackathon_graph() -> StateGraph:
    """
    Build the hackathon mode LangGraph pipeline.
    
    Pipeline phases:
    Phase 0: Data profiling (same as traditional)
    Phase 1: Competition understanding + rubric parsing (NEW)
    Phase 2: Domain + data understanding (same agents, different order)
    Phase 3: Thesis generation + selection (NEW)
    Phase 4: External data acquisition (NEW)
    Phase 5: Hypothesis feature engineering (MODIFIED Feature Factory)
    Phase 6: Model training + validation (same, depth from effort plan)
    Phase 7: Narrative generation (NEW — plots + writeup)
    Phase 8: Deliverable assembly (MODIFIED publisher)
    """
    graph = StateGraph(ProfessorState)
    
    # ══════════════════════════════════════════
    # PHASE 0 — Data Profiling (same as traditional)
    # ══════════════════════════════════════════
    graph.add_node("preflight_checks", _wrap_with_hitl(run_preflight_checks))
    
    # ══════════════════════════════════════════
    # PHASE 1 — Competition Understanding
    # ══════════════════════════════════════════
    graph.add_node("competition_intel", _wrap_with_hitl(competition_intel))
    graph.add_node("rubric_parser", _wrap_with_hitl(run_rubric_parser))
    
    # ══════════════════════════════════════════
    # PHASE 2 — Domain + Data Understanding
    # ══════════════════════════════════════════
    graph.add_node("data_engineer", _wrap_with_hitl(data_engineer))
    graph.add_node("eda_agent", _wrap_with_hitl(eda_agent))
    graph.add_node("domain_research", _wrap_with_hitl(domain_research))
    
    # ══════════════════════════════════════════
    # PHASE 3 — Thesis Generation + Selection
    # ══════════════════════════════════════════
    graph.add_node("thesis_generator", _wrap_with_hitl(thesis_generator))
    
    # ══════════════════════════════════════════
    # PHASE 4 — External Data Acquisition
    # ══════════════════════════════════════════
    graph.add_node("external_data_scout", _wrap_with_hitl(external_data_scout))
    
    # ══════════════════════════════════════════
    # PHASE 5 — Hypothesis Feature Engineering
    # ══════════════════════════════════════════
    # Uses the SAME feature_factory function — it detects hackathon_mode
    # and switches to thesis-testing prompts and Mann-Whitney gates
    graph.add_node("hypothesis_features", _wrap_with_hitl(feature_factory))
    
    # ══════════════════════════════════════════
    # PHASE 6 — Model Training + Validation
    # ══════════════════════════════════════════
    # Only included when effort_plan.technical_depth != "skip"
    # Depth controlled by effort_plan.technical_depth
    graph.add_node("ml_optimizer", _wrap_with_hitl(ml_optimizer))
    graph.add_node("red_team_critic", _wrap_with_hitl(red_team_critic))
    
    # ══════════════════════════════════════════
    # PHASE 7 — Narrative Generation
    # ══════════════════════════════════════════
    graph.add_node("narrative_plots", _wrap_with_hitl(_run_narrative_plots))
    graph.add_node("narrative_writeup", _wrap_with_hitl(_run_narrative_writeup))
    
    # ══════════════════════════════════════════
    # PHASE 8 — Deliverable Assembly
    # ══════════════════════════════════════════
    graph.add_node("hackathon_publisher", _wrap_with_hitl(hackathon_publisher))
    
    # ══════════════════════════════════════════
    # EDGES — The hackathon pipeline order
    # ══════════════════════════════════════════
    
    graph.set_entry_point("preflight_checks")
    
    # Phase 0 → Phase 1
    graph.add_edge("preflight_checks", "competition_intel")
    graph.add_edge("competition_intel", "rubric_parser")
    
    # Phase 1 → Phase 2
    graph.add_edge("rubric_parser", "data_engineer")
    graph.add_edge("data_engineer", "eda_agent")
    graph.add_edge("eda_agent", "domain_research")
    
    # Phase 2 → Phase 3
    # Domain research feeds into thesis generation (thesis needs domain knowledge)
    graph.add_edge("domain_research", "thesis_generator")
    
    # Phase 3 → Phase 4
    # Thesis drives external data search (scout searches for thesis-specific data)
    graph.add_edge("thesis_generator", "external_data_scout")
    
    # Phase 4 → Phase 5
    # External data enriches the dataset before feature engineering
    graph.add_edge("external_data_scout", "hypothesis_features")
    
    # Phase 5 → Phase 6 (conditional — skip model if technical weight is negligible)
    graph.add_conditional_edges(
        "hypothesis_features",
        _route_after_features,
        {
            "train_model": "ml_optimizer",
            "skip_model": "narrative_plots",  # Jump straight to narrative
        }
    )
    
    # Phase 6 → Phase 6 (Critic conditional — replan or continue)
    graph.add_edge("ml_optimizer", "red_team_critic")
    graph.add_conditional_edges(
        "red_team_critic",
        _route_after_critic,
        {
            "replan": "hypothesis_features",  # Replan goes back to features
            "continue": "narrative_plots",    # Continue to narrative
        }
    )
    
    # Phase 7 — Sequential narrative generation
    graph.add_edge("narrative_plots", "narrative_writeup")
    
    # Phase 8 — Final assembly
    graph.add_edge("narrative_writeup", "hackathon_publisher")
    graph.add_edge("hackathon_publisher", END)
    
    return graph
```

### Routing functions

```python
def _route_after_features(state: ProfessorState) -> str:
    """
    Decide whether to train a model or skip to narrative.
    
    Skip model training when:
    - effort_plan says technical_depth is not relevant (extremely rare)
    - No features were generated (thesis features all failed gates)
    
    In practice, model training almost always runs because even
    hackathons with 10% technical weight still need a working notebook.
    """
    effort_plan = state.hackathon_effort_plan or {}
    
    # Check if model training makes sense
    # Even "sprint" technical depth still trains a model — it just uses fewer trials
    # We only skip if there are literally no features to train on
    if not state.feature_manifest and not state.thesis_effect_sizes:
        emit_to_operator(
            "⚠️ No features survived gates. Skipping model training — narrative only.",
            level="STATUS"
        )
        return "skip_model"
    
    return "train_model"


def _route_after_critic(state: ProfessorState) -> str:
    """
    After Critic: replan or continue to narrative.
    Same logic as traditional mode — replan on CONFIRMED_CRITICAL up to MAX_REPLAN_CYCLES.
    """
    verdict = state.critic_verdict
    severity = verdict.get("severity", "CLEAR")
    dag_version = state.dag_version or 0
    
    if severity == "CONFIRMED_CRITICAL" and dag_version < MAX_REPLAN_CYCLES:
        return "replan"
    else:
        return "continue"
```

```python
def _run_narrative_writeup(state: ProfessorState) -> dict:
    """
    LangGraph node wrapper for argumentative writeup generation.
    
    Reads: active_thesis, thesis_effect_sizes, narrative_plots,
           hackathon_rubric, hackathon_effort_plan, hackathon_writeup_template,
           external_datasets, feature_manifest, model_configs, cv_mean,
           ensemble_cv, ensemble_method, domain_brief
    Writes: hackathon_writeup_path, hackathon_writeup_word_count,
            hackathon_writeup_polish_pass
    Emits: STATUS (generating), CHECKPOINT (writeup for operator review)
    """
    from tools.narrative_engine import generate_hackathon_writeup
    
    if not state.hackathon_mode:
        return {}
    
    emit_to_operator("📝 Generating argumentative writeup...", level="STATUS")
    
    session_dir = f"outputs/{state.session_id}"
    
    writeup_path = generate_hackathon_writeup(state, session_dir)
    
    # Read the writeup for word count and operator preview
    word_count = 0
    preview = ""
    try:
        with open(writeup_path, "r") as f:
            content = f.read()
        word_count = len(content.split())
        # Show first 500 chars as preview
        preview = content[:500] + "..." if len(content) > 500 else content
    except Exception:
        preview = "[Could not read writeup]"
    
    max_words = (state.hackathon_rubric or {}).get("writeup_template", {}).get("max_words", 2000)
    effort_plan = state.hackathon_effort_plan or {}
    polish_passes = effort_plan.get("narrative_polish_passes", 2)
    
    msg = (
        f"📝 WRITEUP GENERATED ({word_count} words / {max_words} limit)\n"
        f"Polish passes: {polish_passes}\n\n"
        f"Preview:\n{preview}\n\n"
        f"Reply /continue to proceed, /writeup polish for another pass, "
        f"or /writeup edit \"section\" to regenerate a section"
    )
    
    response = emit_to_operator(msg, level="CHECKPOINT")
    
    # Handle operator requests for additional polish
    additional_polish = 0
    if response and "polish" in (response or ""):
        # Run one more polish pass
        from tools.narrative_engine import _polish_writeup
        with open(writeup_path, "r") as f:
            writeup_content = f.read()
        polished = _polish_writeup(
            writeup_content, 
            state.hackathon_rubric or {}, 
            max_words, 
            1  # One additional pass
        )
        with open(writeup_path, "w") as f:
            f.write(polished)
        word_count = len(polished.split())
        additional_polish = 1
        emit_to_operator(f"📝 Additional polish pass complete. Word count: {word_count}", level="STATUS")
    
    return {
        "hackathon_writeup_path": writeup_path,
        "hackathon_writeup_word_count": word_count,
        "hackathon_writeup_polish_pass": polish_passes + additional_polish,
```

### HITL wrapper (same as traditional graph)

```python
def _wrap_with_hitl(func):
    """
    Decorator that processes HITL queue before running the agent.
    Checks for pause/abort. Processes pending injections.
    Same as traditional graph wrapper.
    """
    def wrapper(state: ProfessorState) -> dict:
        # Check pause/abort
        if state.pipeline_paused:
            emit_to_operator("⏸️ Pipeline paused. Send /resume.", level="STATUS")
            # Block until resumed (handled by HITL infrastructure)
        if state.pipeline_aborted:
            raise PipelineAborted("Pipeline aborted by operator")
        
        # Process pending injections
        state = process_pending_injections(state)
        
        # Run the actual agent
        return func(state)
    
    wrapper.__name__ = func.__name__
    return wrapper
```

### What's NOT in the hackathon graph (excluded from traditional)

These traditional agents are EXCLUDED because they don't serve the hackathon objective:

| Excluded Agent | Why |
|---|---|
| `metric_verification_gate` | Only useful if there's a leaderboard metric. Most hackathons are writeup-judged. |
| `shift_detector` | Distribution shift matters for LB accuracy but not for narrative quality. |
| `validation_architect` | CV strategy still runs inside ML Optimizer. Separate architect step not needed. |
| `problem_reframer` | The THESIS is the reframing. No separate reframer needed. |
| `creative_hypothesis` | Thesis features replace creative features. Same concept, different prompt. |
| `self_reflection` | Rules from past competitions still load (from ChromaDB), but no new rules generated during hackathon. |
| `pseudo_label` | Semi-supervised not relevant for hackathon deliverables. |
| `ensemble_architect` | Ensemble is overkill for hackathon. Best single model is sufficient for the 30% technical score. |
| `post_processor` | Minimal value in hackathon context. |
| `submission_strategist` | No leaderboard submission strategy needed. |

**Note:** The ML Optimizer still runs its own internal CV strategy. It doesn't need the Validation Architect as a separate node — it reads `state.validation_strategy` which is populated by Pre-Flight or defaults.

```python
def run_hackathon(
    competition_name: str,
    data_dir: str,
    competition_url: str = "",
    hitl_channels: list = None,
    hitl_mode: str = "supervised",
    llm_budget_calls: int = 200,  # Hackathon default slightly higher
    llm_budget_usd: float = 10.0,
) -> ProfessorState:
    """
    Run Professor in hackathon mode.
    
    Usage:
        professor run --mode hackathon --competition triagegeist --data ./data
    """
    from langgraph.checkpoint.sqlite import SqliteSaver
    
    # Build the graph
    graph = build_hackathon_graph()
    
    # Compile with checkpointer
    checkpointer = SqliteSaver.from_conn_string("hackathon_checkpoints.db")
    app = graph.compile(checkpointer=checkpointer)
    
    # Initial state
    initial_state = ProfessorState(
        session_id=f"hackathon_{competition_name}_{datetime.utcnow().strftime('%Y%m%d_%H%M%S')}",
        competition_name=competition_name,
        competition_url=competition_url,
        raw_data_path=data_dir,
        hackathon_mode=True,  # This flag switches Feature Factory to thesis mode
        hitl_mode=hitl_mode,
        hitl_channels=hitl_channels or ["cli"],
        llm_budget_calls_max=llm_budget_calls,
        llm_budget_usd_max=llm_budget_usd,
    )
    
    # Initialize HITL
    from tools.operator_channel import init_hitl
    init_hitl(channels=initial_state.hitl_channels, config={})
    
    # Initialize Cost Governor
    from shields.cost_governor import init_cost_governor
    init_cost_governor(max_calls=llm_budget_calls, max_usd=llm_budget_usd)
    
    # Run
    config = {"configurable": {"thread_id": initial_state.session_id}}
    
    emit_to_operator(
        f"🎯 HACKATHON MODE — {competition_name}\n"
        f"Pipeline: thesis-driven (narrative is the deliverable)\n"
        f"HITL: {hitl_mode} on {hitl_channels}\n"
        f"Budget: {llm_budget_calls} calls / ${llm_budget_usd:.2f}",
        level="STATUS"
    )
    
    final_state = app.invoke(initial_state, config=config)
    
    return final_state
```

## COMMIT 1B: agents/hackathon_publisher.py

### What this does

The final node in the hackathon graph. Assembles ALL required deliverables based on `rubric.submission_requirements`. Produces whatever the competition asks for: notebook, writeup, cover image, project link README, requirements.txt.

```python
"""
Hackathon Publisher — assembles all required deliverables.

Unlike the traditional publisher which produces submission.csv + notebook + writeup,
the hackathon publisher produces whatever the competition rubric requires:
- Kaggle Notebook (.ipynb format for hackathons, not .py)
- Project Writeup (already generated by Narrative Engine)
- Cover Image (auto-generated from the key finding visualization)
- Project Link README (for GitHub)
- requirements.txt
"""

import os
import json
import shutil
from datetime import datetime
from graph.state import ProfessorState
from tools.operator_channel import emit_to_operator
from tools.code_ledger import get_kept_entries, get_reasoning_chain
from shields.submission_safety import verify_submission, backup_submission


def hackathon_publisher(state: ProfessorState) -> dict:
    """
    Final hackathon pipeline node. Produces all required deliverables.
    
    Reads: EVERYTHING — needs full state for notebook + writeup + verification
    Writes: solution_notebook_path, solution_writeup_path, code_ledger_path,
            notebook_reproduction_validated
    Emits: RESULT (Milestone — final summary), GATE (GUIDED mode: submit?)
    """
    session_dir = f"outputs/{state.session_id}"
    os.makedirs(session_dir, exist_ok=True)
    
    rubric = state.hackathon_rubric or {}
    requirements = rubric.get("submission_requirements", ["notebook", "writeup"])
    
    deliverables = {}
    errors = []
    
    emit_to_operator("📦 Assembling hackathon deliverables...", level="STATUS")
    
    # ═══════════════════════════════════════
    # 1. NOTEBOOK (always required)
    # ═══════════════════════════════════════
    if "notebook" in requirements:
        try:
            notebook_path = _assemble_hackathon_notebook(state, session_dir)
            deliverables["notebook"] = notebook_path
            emit_to_operator(f"✅ Notebook generated: {notebook_path}", level="STATUS")
        except Exception as e:
            errors.append(f"Notebook generation failed: {str(e)[:200]}")
            emit_to_operator(f"❌ Notebook generation failed: {str(e)[:100]}", level="STATUS")
    
    # ═══════════════════════════════════════
    # 2. WRITEUP (already generated by Narrative Engine)
    # ═══════════════════════════════════════
    if "writeup" in requirements:
        writeup_path = state.hackathon_writeup_path
        if writeup_path and os.path.exists(writeup_path):
            deliverables["writeup"] = writeup_path
            emit_to_operator(f"✅ Writeup ready: {writeup_path} ({state.hackathon_writeup_word_count} words)", level="STATUS")
        else:
            errors.append("Writeup not generated by Narrative Engine")
            emit_to_operator("❌ Writeup missing — Narrative Engine may have failed", level="STATUS")
    
    # ═══════════════════════════════════════
    # 3. COVER IMAGE (if required)
    # ═══════════════════════════════════════
    if "cover_image" in requirements:
        try:
            cover_path = _generate_cover_image(state, session_dir)
            deliverables["cover_image"] = cover_path
            emit_to_operator(f"✅ Cover image generated: {cover_path}", level="STATUS")
        except Exception as e:
            errors.append(f"Cover image failed: {str(e)[:200]}")
            emit_to_operator(f"⚠️ Cover image failed — use a narrative plot as cover", level="STATUS")
    
    # ═══════════════════════════════════════
    # 4. PROJECT LINK / README (if required)
    # ═══════════════════════════════════════
    if "project_link" in requirements:
        try:
            readme_path = _generate_github_readme(state, session_dir)
            deliverables["readme"] = readme_path
            emit_to_operator(
                f"✅ README.md generated for GitHub repo\n"
                f"⚠️ You need to: create a GitHub repo, push the code, paste the URL into the submission",
                level="STATUS"
            )
        except Exception as e:
            errors.append(f"README failed: {str(e)[:200]}")
    
    # ═══════════════════════════════════════
    # 5. REQUIREMENTS.TXT (always)
    # ═══════════════════════════════════════
    try:
        req_path = _generate_requirements(deliverables.get("notebook", ""), session_dir)
        deliverables["requirements"] = req_path
    except Exception:
        pass  # Non-critical
    
    # ═══════════════════════════════════════
    # 6. SUBMISSION VERIFICATION (if there's a submission.csv)
    # ═══════════════════════════════════════
    submission_path = state.submission_path or os.path.join(session_dir, "submission.csv")
    if os.path.exists(submission_path):
        verification = verify_submission(
            submission_path=submission_path,
            sample_submission_path=state.preflight_submission_format.get("path", ""),
            canonical_test_rows=state.canonical_test_rows,
        )
        if verification["valid"]:
            backup_submission(submission_path, session_dir)
            deliverables["submission"] = submission_path
        else:
            errors.append(f"Submission verification failed: {verification['issues']}")
    
    # ═══════════════════════════════════════
    # 7. CODE LEDGER (always — provenance)
    # ═══════════════════════════════════════
    ledger_path = os.path.join(session_dir, "code_ledger.jsonl")
    if os.path.exists(ledger_path):
```

### Hackathon notebook assembly

Hackathon notebooks are different from traditional solution_notebook.py:
- Format: `.ipynb` (Jupyter/Kaggle format), NOT `.py`
- Content: INTERLEAVED markdown + code cells (the notebook IS part of the narrative)
- Audience: Judges read it. It must be clear, well-commented, and tell a story.

```python
def _assemble_hackathon_notebook(state: ProfessorState, session_dir: str) -> str:
    """
    Assemble a Kaggle-compatible .ipynb notebook.
    
    Unlike traditional solution_notebook.py:
    - Format is .ipynb (JSON structure with markdown + code cells)
    - Interleaves narrative markdown with executable code
    - The notebook IS part of the submission — judges read it
    - Must run end-to-end without errors
    """
    cells = []
    
    # ── Title + Introduction Cell (markdown) ──
    thesis = state.active_thesis or {}
    cells.append(_markdown_cell(
        f"# {thesis.get('statement', state.competition_name)}\n\n"
        f"**Competition:** {state.competition_name}\n"
        f"**Approach:** {thesis.get('angle', 'Thesis-driven analysis')}\n"
        f"**Target Audience:** {thesis.get('target_audience', 'Practitioners')}\n\n"
        f"This notebook implements and validates our thesis through structured "
        f"data analysis, feature engineering, and model training."
    ))
    
    # ── Setup Cell (code) ──
    cells.append(_code_cell(
        "# Setup\n"
        "import polars as pl\n"
        "import numpy as np\n"
        "import matplotlib.pyplot as plt\n"
        "import seaborn as sns\n"
        "from sklearn.model_selection import StratifiedKFold\n"
        "from sklearn.metrics import roc_auc_score\n"
        "import warnings\n"
        "warnings.filterwarnings('ignore')\n"
        "\n"
        "SEED = 42\n"
        "np.random.seed(SEED)\n"
        "sns.set_style('whitegrid')\n"
        "plt.rcParams['figure.figsize'] = (10, 6)\n"
        "plt.rcParams['font.size'] = 12"
    ))
    
    # ── Data Loading Section ──
    cells.append(_markdown_cell("## Data Loading & Exploration"))
    
    # Get kept code blocks from Code Ledger, grouped by agent
    kept_entries = get_kept_entries(session_dir)
    
    # Data Engineer code
    de_entries = [e for e in kept_entries if e["agent"] == "data_engineer" and e["success"]]
    for entry in de_entries:
        cells.append(_markdown_cell(f"### {entry['purpose']}"))
        cells.append(_code_cell(_clean_code(entry["code"])))
    
    # EDA code + narrative
    cells.append(_markdown_cell("## Exploratory Data Analysis"))
    eda_entries = [e for e in kept_entries if e["agent"] == "eda_agent" and e["success"]]
    for entry in eda_entries:
        cells.append(_code_cell(_clean_code(entry["code"])))
    
    # ── Thesis Statement ──
    cells.append(_markdown_cell(
        f"## Our Thesis\n\n"
        f"> **{thesis.get('statement', '')}**\n\n"
        f"**Hypothesis:** {thesis.get('hypothesis', '')}\n\n"
        f"**Condition:** We compare outcomes across `{thesis.get('condition_variable', '')}` "
        f"to test whether this factor creates systematic differences."
    ))
    
    # ── External Data Integration ──
    if state.external_datasets:
        cells.append(_markdown_cell("## External Data Integration"))
        scout_entries = [e for e in kept_entries if e["agent"] == "external_data_scout" and e["success"]]
        for entry in scout_entries:
            cells.append(_markdown_cell(f"### {entry['purpose']}"))
            cells.append(_code_cell(_clean_code(entry["code"])))
    
    # ── Feature Engineering ──
    cells.append(_markdown_cell("## Hypothesis-Testing Feature Engineering"))
    ff_entries = [e for e in kept_entries if e["agent"] == "feature_factory" and e["success"]]
    for entry in ff_entries:
        cells.append(_markdown_cell(f"### Round {entry.get('round_num', '?')}: {entry['purpose']}"))
        cells.append(_code_cell(_clean_code(entry["code"])))
    
    # ── Results — Effect Sizes ──
    if state.thesis_effect_sizes:
        cells.append(_markdown_cell("## Results: Testing the Thesis"))
        
        effect_summary = "| Feature | Effect Size | P-value | Direction |\n|---|---|---|---|\n"
        for fname, effect in state.thesis_effect_sizes.items():
            effect_summary += (
                f"| {fname} | {effect.get('effect_size', 'N/A'):.3f} "
                f"({effect.get('effect_magnitude', '?')}) | "
                f"{effect.get('p_value', 'N/A'):.4f} | "
                f"{effect.get('direction', '?')} |\n"
            )
        cells.append(_markdown_cell(effect_summary))
    
    # ── Model Training ──
    cells.append(_markdown_cell("## Model Training"))
    ml_entries = [e for e in kept_entries if e["agent"] == "ml_optimizer" and e["success"]]
    for entry in ml_entries:
        cells.append(_code_cell(_clean_code(entry["code"])))
    
    # ── Narrative Plots ──
    if state.narrative_plots:
        cells.append(_markdown_cell("## Key Findings — Visualizations"))
        for plot in state.narrative_plots:
            if plot.get("path") and os.path.exists(plot["path"]):
                cells.append(_markdown_cell(f"### {plot.get('caption', 'Figure')}"))
                # Embed plot as inline image (copy to notebook output directory)
                plot_filename = os.path.basename(plot["path"])
                cells.append(_code_cell(
                    f"from IPython.display import Image, display\n"
                    f"display(Image(filename='{plot_filename}'))"
                ))
    
    # ── Conclusion ──
    cells.append(_markdown_cell(
        "## Conclusion\n\n"
        f"Our analysis {'supports' if state.thesis_effect_sizes else 'explored'} "
        f"the thesis that {thesis.get('statement', 'N/A')}. "
        f"The key findings are documented in the project writeup."
    ))
    
    # ── Build .ipynb structure ──
    notebook = {
        "metadata": {
            "kernelspec": {
                "display_name": "Python 3",
                "language": "python",
                "name": "python3"
            },
            "language_info": {
                "name": "python",
                "version": "3.11.0"
            }
        },
        "nbformat": 4,
        "nbformat_minor": 5,
        "cells": cells,
    }
    
    notebook_path = os.path.join(session_dir, "hackathon_notebook.ipynb")
    with open(notebook_path, "w") as f:
        json.dump(notebook, f, indent=2)
    
    return notebook_path


def _markdown_cell(content: str) -> dict:
    return {
        "cell_type": "markdown",
        "metadata": {},
        "source": [content],
    }


def _code_cell(content: str) -> dict:
    return {
        "cell_type": "code",
        "metadata": {},
        "source": [content],
        "outputs": [],
        "execution_count": None,
    }


def _clean_code(code: str) -> str:
    """Clean sandbox code for notebook inclusion."""
    import re
    lines = code.split("\n")
    cleaned = []
    skip_patterns = [
        "from tools.", "from agents.", "from shields.", "from graph.",
        "emit_to_operator(", "run_in_sandbox(", "mark_rejected(",
        "__DIAGNOSTICS__", "# === USER CODE",
    ]
    for line in lines:
        if any(pat in line.strip() for pat in skip_patterns):
            continue
        # Replace absolute sandbox paths
        line = re.sub(r'["\'].*?/sandbox/.*?/([^/"\']+)["\']', r'"\1"', line)
        cleaned.append(line)
    return "\n".join(cleaned)
```

def _generate_cover_image(state: ProfessorState, session_dir: str) -> str:
    """
    Generate a cover image from the key finding visualization.
    Resizes the best narrative plot to competition dimensions.
    
    Triagegeist requires 560 x 280 px.
    """
    from tools.sandbox import run_in_sandbox
    
    # Find the best narrative plot (finding_summary or conditional_comparison)
    source_plot = None
    for plot in (state.narrative_plots or []):
        if plot.get("path") and os.path.exists(plot["path"]):
            if plot.get("plot_type") in ("finding_summary", "conditional_comparison"):
                source_plot = plot["path"]
                break
    
    if not source_plot:
        # Use the first available plot
        for plot in (state.narrative_plots or []):
            if plot.get("path") and os.path.exists(plot["path"]):
                source_plot = plot["path"]
                break
    
    if not source_plot:
        raise ValueError("No narrative plots available for cover image")
    
    # Resize to competition dimensions
    cover_path = os.path.join(session_dir, "cover_image.png")
    
    code = f"""
from PIL import Image

img = Image.open("{source_plot}")
# Crop to 2:1 aspect ratio (560x280)
width, height = img.size
target_ratio = 560 / 280  # 2:1
current_ratio = width / height

if current_ratio > target_ratio:
    # Too wide — crop sides
    new_width = int(height * target_ratio)
    left = (width - new_width) // 2
    img = img.crop((left, 0, left + new_width, height))
else:
    # Too tall — crop top/bottom
    new_height = int(width / target_ratio)
    top = (height - new_height) // 2
    img = img.crop((0, top, width, top + new_height))

img = img.resize((560, 280), Image.LANCZOS)
img.save("{cover_path}")
print(f"Cover image saved: {{img.size}}")
"""
    
    result = run_in_sandbox(code=code, agent_name="hackathon_publisher", purpose="Generate cover image")
    
    if result["success"] and os.path.exists(cover_path):
        return cover_path
    else:
        raise ValueError(f"Cover image generation failed: {result['stderr'][:200]}")

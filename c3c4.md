```python
from dataclasses import dataclass, field
from typing import Optional

@dataclass
class DatasetCandidate:
    """A candidate external dataset discovered during scouting."""
    name: str
    source_url: str
    source_type: str                # "competition_recommended" | "kaggle" | "government" | "academic" | "domain_specific" | "reference_table"
    description: str
    
    # Feasibility dimensions (0.0-1.0 unless bool)
    join_feasibility: float         # Can it be joined to competition data?
    relevance_to_thesis: float      # Does it help test the active thesis?
    size_compatible: bool           # Fits in memory alongside competition data?
    license_compatible: bool        # License allows use in competition?
    download_accessible: bool       # Can we actually get it programmatically?
    
    # Join details
    join_key: str                   # Column to join on: "age_group", "icd_code", "date"
    join_type: str                  # "left" | "inner" | "lookup_table" | "enrichment"
    estimated_match_rate: float     # % of competition rows that will match (0.0-1.0)
    
    # Computed
    overall_score: float            # Weighted combination: 0.4*relevance + 0.3*join_feasibility + 0.2*match_rate + 0.1*accessibility
    
    # Integration status
    downloaded: bool = False
    download_path: str = ""
    integrated: bool = False
    integration_result: dict = field(default_factory=dict)
```

```python
def _parse_data_needs(active_thesis: dict) -> list[str]:
    """
    Extract specific data needs from the thesis's data_plan.
    
    The thesis says: external_needed: ["AHA cardiac risk thresholds by age", 
                                        "ESI inter-rater reliability studies"]
    
    Convert each need into search queries.
    """
    data_plan = active_thesis.get("data_plan", {})
    external_needed = data_plan.get("external_needed", [])
    
    if not external_needed:
        return []
    
    search_queries = []
    for need in external_needed:
        # Each need becomes 2-3 search queries targeting different sources
        queries = [
            f"{need} dataset download CSV",
            f"{need} kaggle dataset",
            f"{need} public data",
        ]
        search_queries.extend(queries)
    
    return search_queries
```

```python
def _search_all_sources(
    queries: list[str],
    recommended_datasets: list[dict],
    domain: str,
    effort_priority: str,
) -> list[DatasetCandidate]:
    """
    Search all source tiers for candidate datasets.
    
    Args:
        queries: Search queries derived from thesis data needs
        recommended_datasets: From rubric.recommended_datasets
        domain: Domain classification for domain-specific searches
        effort_priority: "high" | "medium" | "skip" from effort plan
    
    Returns: List of DatasetCandidate objects (unevaluated — scores all 0)
    """
    candidates = []
    
    # Tier 1: Competition-recommended datasets (always check)
    for rec in recommended_datasets:
        candidates.append(DatasetCandidate(
            name=rec.get("name", "Unknown"),
            source_url=rec.get("url", ""),
            source_type="competition_recommended",
            description=rec.get("description", ""),
            join_feasibility=0.0,  # Evaluated in Step 3
            relevance_to_thesis=0.0,
            size_compatible=True,  # Assume yes for recommended
            license_compatible=True,  # Recommended = approved
            download_accessible=True,
            join_key="",
            join_type="",
            estimated_match_rate=0.0,
            overall_score=0.0,
        ))
    
    if effort_priority == "skip":
        return candidates  # Only check recommended, skip web search
    
    # Tier 2: Kaggle Datasets API (if available)
    kaggle_results = _search_kaggle_datasets(queries)
    candidates.extend(kaggle_results)
    
    # Tier 3-6: Web search (if effort_priority is "high" or "medium")
    if effort_priority == "high":
        # Search all tiers
        web_results = _search_web_sources(queries, domain)
        candidates.extend(web_results)
    elif effort_priority == "medium":
        # Search only government and academic
        web_results = _search_web_sources(queries, domain, tiers=["government_portals", "academic_repositories"])
        candidates.extend(web_results)
    
    # Deduplicate by name (case-insensitive)
    seen_names = set()
    deduped = []
    for c in candidates:
        name_key = c.name.lower().strip()
        if name_key not in seen_names:
            seen_names.add(name_key)
            deduped.append(c)
    
    return deduped
```

```python
def _search_kaggle_datasets(queries: list[str]) -> list[DatasetCandidate]:

import json

result = subprocess.run(
    ["kaggle", "datasets", "list", "-s", "{query}", "--csv", "--max-size", "500000000"],
    capture_output=True, text=True, timeout=30
)

if result.returncode == 0:
    lines = result.stdout.strip().split("\\n")
    if len(lines) > 1:  # Header + results
        for line in lines[1:6]:  # Top 5 results
            parts = line.split(",")

            code=code,

        if result["success"]:

                        join_key="",
                        join_type="",

                    ))
                except json.JSONDecodeError:
                    continue
```

### Step 3 — Evaluate each candidate (LLM)

```python
def _evaluate_candidates(
    candidates: list[DatasetCandidate],
    thesis: dict,
    competition_schema: dict,
) -> list[DatasetCandidate]:
    """
    Use LLM to evaluate each candidate's relevance and join feasibility.
    Batch evaluate to save LLM calls — all candidates in one prompt.
    """
    from tools.llm_provider import llm_call
    
    if not candidates:
        return []
    
    # Build candidate descriptions for the prompt
    candidate_descriptions = "\n".join([
        f"{i+1}. {c.name} — {c.description[:200]} (source: {c.source_type}, url: {c.source_url[:100]})"
        for i, c in enumerate(candidates[:15])  # Max 15 to fit in context
    ])
    
    schema_summary = ", ".join([f"{col}: {dtype}" for col, dtype in list(competition_schema.items())[:30]])
    
    prompt = f"""Evaluate these external dataset candidates for a hackathon thesis.

THESIS: "{thesis.get('statement', '')}"
THESIS DATA NEEDS: {thesis.get('data_plan', {}).get('external_needed', [])}
CONDITION VARIABLE: {thesis.get('condition_variable', '')}

COMPETITION DATA SCHEMA (columns to join on):
{schema_summary}

CANDIDATE DATASETS:
{candidate_descriptions}

For each candidate, evaluate:
1. RELEVANCE (0.0-1.0): Does this dataset DIRECTLY help test the thesis? 
   1.0 = essential for the thesis. 0.5 = somewhat related. 0.0 = irrelevant.
2. JOIN_FEASIBILITY (0.0-1.0): Can this dataset be joined to the competition data?
   1.0 = obvious shared key. 0.5 = requires transformation. 0.0 = no common key.
3. JOIN_KEY: What column(s) to join on. Use competition schema column names.
4. JOIN_TYPE: "left" (keep all comp rows) | "lookup_table" (reference data) | "enrichment" (add columns)
5. MATCH_RATE (0.0-1.0): What fraction of competition rows would have a match?
6. LICENSE: Is this dataset freely usable in a competition? true/false
7. SIZE_OK: Would this dataset fit in memory (<500MB)? true/false

Respond with ONLY valid JSON array:
[
    {{
        "candidate_index": 1,
        "relevance": 0.8,
        "join_feasibility": 0.7,
        "join_key": "age_group",
        "join_type": "lookup_table",
        "match_rate": 0.95,
        "license_ok": true,
        "size_ok": true,
        "reasoning": "This dataset contains age-specific cardiac thresholds that directly enable testing the thesis condition"
    }},
    ...
]

Skip candidates with relevance < 0.3 — don't include them in the response.
"""
    
    response = llm_call(
        prompt=prompt,
        agent_name="external_data_scout",
        temperature=0.2,  # Accurate evaluation, not creative
        response_format="json",
    )
    
    # Parse and apply evaluations
    evaluations = _safe_json_parse_list(response["text"])
    
    for eval_entry in evaluations:
        idx = eval_entry.get("candidate_index", 0) - 1
        if 0 <= idx < len(candidates):
            c = candidates[idx]
            c.relevance_to_thesis = eval_entry.get("relevance", 0.0)
            c.join_feasibility = eval_entry.get("join_feasibility", 0.0)
            c.join_key = eval_entry.get("join_key", "")
            c.join_type = eval_entry.get("join_type", "left")
            c.estimated_match_rate = eval_entry.get("match_rate", 0.0)
            c.license_compatible = eval_entry.get("license_ok", True)
            c.size_compatible = eval_entry.get("size_ok", True)
            
            # Compute overall score
            c.overall_score = (
                0.40 * c.relevance_to_thesis +
                0.30 * c.join_feasibility +
                0.20 * c.estimated_match_rate +
                0.10 * (1.0 if c.download_accessible else 0.0)
            )
    
    # Sort by overall score descending
    candidates.sort(key=lambda c: c.overall_score, reverse=True)
    
    return candidates
```

```python
def _present_candidates(candidates: list[DatasetCandidate], thesis: dict) -> str:
```

```python
def _integrate_dataset(
    candidate: DatasetCandidate,
    competition_data_path: str,
    session_dir: str,
    canonical_train_rows: int,
) -> dict:
    """
    Download external dataset and join to competition data.
    Executed via run_in_sandbox() with full safety infrastructure.
    
    Returns:
    {
        "success": bool,
        "enriched_path": str,    # Path to joined data
        "new_columns": list,     # Columns added from external data
        "matched_rows": int,     # Rows that matched the join
        "match_rate": float,     # matched_rows / canonical_train_rows
        "error": str or None,
    }
    """
    from tools.sandbox import run_in_sandbox
    from tools.llm_provider import llm_call
    
    # Generate join code via LLM
    join_prompt = f"""Write Polars code to integrate an external dataset with competition data.

COMPETITION DATA: load from "{competition_data_path}"
EXTERNAL DATA: load from "{candidate.download_path}"
JOIN KEY: {candidate.join_key}
JOIN TYPE: {candidate.join_type}
EXPECTED ROW COUNT AFTER JOIN: {canonical_train_rows} (MUST be preserved exactly)

Requirements:
1. Load both datasets with Polars (import polars as pl)
2. Examine the external dataset — print columns, shape, first 3 rows
3. Join on {candidate.join_key} using {candidate.join_type} join
4. VERIFY: output row count == {canonical_train_rows}. If not, debug the join.
5. If join duplicates rows, use a left join with unique external keys only
6. Save enriched dataset to "{session_dir}/enriched_data.parquet"
7. Print JSON summary: {{"matched_rows": int, "new_columns": list, "total_rows": int}}

CRITICAL: The competition data's row count MUST be preserved. 
If the join changes the row count, it's WRONG. Debug and fix.
Use Polars, not Pandas.
"""
    
    response = llm_call(
        prompt=join_prompt,
        agent_name="external_data_scout",
        temperature=0.2,
    )
    
    code = _extract_code(response["text"])
    
    result = run_in_sandbox(
        code=code,
        agent_name="external_data_scout",
        purpose=f"Integrate external dataset: {candidate.name}",
        expected_row_change="none",  # Row count MUST be preserved
    )
    
    if result["success"]:
        # Parse the summary from stdout
        summary = _parse_integration_summary(result["stdout"])
        
        # Verify row count preservation
        enriched_path = f"{session_dir}/enriched_data.parquet"
        if os.path.exists(enriched_path):
            import polars as pl
            enriched = pl.read_parquet(enriched_path)
            if len(enriched) != canonical_train_rows:
                return {
                    "success": False,
                    "error": f"Row count changed: {canonical_train_rows} → {len(enriched)}. Join corrupted data.",
                    "enriched_path": "",
                    "new_columns": [],
                    "matched_rows": 0,
                    "match_rate": 0.0,
                }
        
        return {
            "success": True,
            "enriched_path": enriched_path,
            "new_columns": summary.get("new_columns", []),
            "matched_rows": summary.get("matched_rows", 0),
            "match_rate": summary.get("matched_rows", 0) / max(canonical_train_rows, 1),
            "error": None,
        }
    else:
        return {
            "success": False,
            "error": result["stderr"][:500],
            "enriched_path": "",
            "new_columns": [],
            "matched_rows": 0,
            "match_rate": 0.0,
        }
```

```python
def _construct_reference_table(

    domain: str,
    session_dir: str,

    """
    When the thesis needs reference thresholds or clinical guidelines,
    construct them as a lookup table from LLM knowledge + web search.
    
    Example: "AHA cardiac risk thresholds by age" → 
    DataFrame with columns: age_min, age_max, hr_threshold, bp_threshold, risk_level
    """
    from tools.llm_provider import llm_call
    from tools.sandbox import run_in_sandbox
    
    # Ask LLM to generate the reference table
    prompt = f"""You are a senior {domain} practitioner.

Generate a reference lookup table for: "{data_need}"

The table must be:
- Based on published, widely-accepted standards (cite the source)
- Structured as a CSV with clear column headers
- Complete enough to be joined to a patient/event dataset

Respond with:
1. SOURCE: What published guideline this is from (e.g., "AHA 2019 Guidelines")
2. CSV: The complete table in CSV format
3. JOIN_KEY: What column to join on (e.g., "age_group")

Format:
SOURCE: <source>
CSV:
col1,col2,col3
val1,val2,val3
...
JOIN_KEY: <key>
"""
    
    response = llm_call(prompt=prompt, agent_name="external_data_scout", temperature=0.2)

        join_key=join_key,
        join_type="lookup_table",
```

### The main node function

```python
def external_data_scout(state: ProfessorState) -> dict:
    from tools.operator_channel import emit_to_operator
    
    # Gate checks
    if not state.hackathon_mode:
        return {}
    
    thesis = state.active_thesis
    if not thesis:
        emit_to_operator("⚠️ No active thesis — skipping external data scout", level="STATUS")
        return {"thesis_data_sufficient": True}
    
    effort_plan = state.hackathon_effort_plan or {}
    effort_priority = effort_plan.get("external_data_priority", "medium")
    
    data_needs = thesis.get("data_plan", {}).get("external_needed", [])
    if not data_needs:
        emit_to_operator("📁 No external data needed for this thesis", level="STATUS")
        return {"thesis_data_sufficient": True, "external_datasets": [], "enriched_data_path": state.clean_data_path}
    
    emit_to_operator(f"📁 Searching for external data: {data_needs}", level="STATUS")
    
    # Step 1: Parse needs into queries
    queries = _parse_data_needs(thesis)
    
    # Step 2: Search all sources
    recommended = state.hackathon_rubric.get("recommended_datasets", [])
    candidates = _search_all_sources(queries, recommended, state.domain_classification or "general", effort_priority)
    
    # Step 2b: Construct reference tables for threshold/guideline needs
    for need in data_needs:
        need_lower = need.lower()
        if any(kw in need_lower for kw in ["threshold", "guideline", "protocol", "standard", "criteria", "reference", "lookup"]):
            ref_table = _construct_reference_table(need, state.domain_classification or "general", 
                                                     f"outputs/{state.session_id}")
            candidates.append(ref_table)
    
    # Step 3: Evaluate candidates
    if candidates:
        candidates = _evaluate_candidates(candidates, thesis, state.data_schema or {})
    
    # Step 4: Present to operator
    response = _present_candidates(candidates, thesis)
    
    # Parse operator response
    approved_indices = _parse_approval(response, candidates)
    
    # Step 5: Download and integrate top candidates (max 3)
    MAX_EXTERNAL = 3
    approved = [candidates[i] for i in approved_indices if i < len(candidates)][:MAX_EXTERNAL]
    
    integrated_datasets = []
    enriched_path = state.clean_data_path  # Start with original data
    
    for candidate in approved:
        emit_to_operator(f"📁 Integrating: {candidate.name}...", level="STATUS")
        
        # Download if not already downloaded (reference tables are pre-downloaded)
        if not candidate.downloaded:
            candidate = _download_dataset(candidate, f"outputs/{state.session_id}")
        
        if candidate.downloaded:
            result = _integrate_dataset(
                candidate=candidate,
                competition_data_path=enriched_path,  # Chain joins
                session_dir=f"outputs/{state.session_id}",
                canonical_train_rows=state.canonical_train_rows,
            )
            
            if result["success"]:
                enriched_path = result["enriched_path"]
                candidate.integrated = True
                candidate.integration_result = result
                integrated_datasets.append(_candidate_to_dict(candidate))
                emit_to_operator(
                    f"✅ {candidate.name}: {result['matched_rows']} rows matched "
                    f"({result['match_rate']:.0%}), {len(result['new_columns'])} new columns",
                    level="STATUS"
                )
            else:
                emit_to_operator(
                    f"❌ {candidate.name}: integration failed — {result['error'][:100]}",
                    level="STATUS"
                )
    
    # Determine if we have enough data
    thesis_data_sufficient = len(integrated_datasets) > 0 or effort_priority == "skip"
    
    if not thesis_data_sufficient:
        emit_to_operator(
            "⚠️ No external data integrated. Analysis will proceed with competition data only.\n"
            "The writeup will note this as a limitation and suggest future data sources.",
            level="STATUS"
        )
    
    return {
        "external_datasets": integrated_datasets,
        "external_data_paths": [d.get("download_path", "") for d in integrated_datasets],
        "thesis_data_sufficient": thesis_data_sufficient,
        "enriched_data_path": enriched_path,
    }


def _parse_approval(response: str, candidates: list) -> list[int]:
    """Parse operator's dataset approval response."""
    if response is None or "/continue" in (response or ""):
        # Default: approve top 3 with score > 0.3
        return [i for i, c in enumerate(candidates) if c.overall_score > 0.3][:3]
    
    if "/data approve" in (response or ""):
        # Parse specific indices: /data approve 1,3
        parts = response.split("approve")[-1].strip()
        try:
            indices = [int(x.strip()) - 1 for x in parts.split(",")]
            return [i for i in indices if 0 <= i < len(candidates)]
        except ValueError:
            return [i for i, c in enumerate(candidates) if c.overall_score > 0.3][:3]
    
    return [i for i, c in enumerate(candidates) if c.overall_score > 0.3][:3]


def _candidate_to_dict(candidate: DatasetCandidate) -> dict:
    from dataclasses import asdict
    return asdict(candidate)
```

## COMMITS 2-3: Narrative Engine (tools/narrative_engine.py)

### What this does

Two capabilities:
- **Part A (Commit 2):** Generates 3-7 argument-supporting visualizations that make the thesis finding visually undeniable
- **Part B (Commit 3):** Generates a template-aware argumentative writeup as the PRIMARY deliverable

### The key difference from EDA Artifact Export

EDA plots ask "what does this data look like?" (diagnostic, for the operator).
Narrative plots ask "does this data support the thesis?" (argumentative, for the judges).

EDA writeup documents what Professor DID (process description).
Narrative writeup argues for what Professor FOUND (persuasive argument).

```python
def _select_plot_types(thesis: dict, n_plots: int) -> list[str]:
    """
    Choose plot types based on what the thesis needs to prove.
    Not random — each type serves a specific argumentative purpose.
    """
    condition_type = thesis.get("condition_type", "categorical")
    
    # Required plots (every thesis needs these)
    plots = [
        "conditional_comparison",   # The core finding — groups compared
        "finding_summary",          # Infographic summarizing the thesis result
    ]
    
    # Condition-type-specific plots
    if condition_type == "categorical":
        plots.insert(1, "grouped_distribution")  # Side-by-side histograms by condition
    elif condition_type == "temporal":
        plots.insert(1, "temporal_effect")  # Time series of the effect
    elif condition_type == "spatial":
        plots.insert(1, "spatial_heatmap")  # Geographic distribution of effect
    elif condition_type == "threshold":
        plots.insert(1, "threshold_curve")  # Effect as a function of threshold value
    elif condition_type == "interaction":
        plots.insert(1, "interaction_heatmap")  # 2D heatmap of moderator × condition
    
    # If thesis involves model comparison (most hackathons do)
    if thesis.get("estimated_scores", {}).get("Technical Quality", {}).get("score", 0) > 20:
        plots.append("model_comparison")  # ROC curve or calibration plot
    
    # Fill remaining slots
    supplementary = ["effect_by_subgroup", "feature_importance_thesis", "supporting_scatter"]
    while len(plots) < n_plots:
        if supplementary:
            plots.append(supplementary.pop(0))
        else:
            plots.append("supporting_evidence")
    
    return plots[:n_plots]
```

```python
def _get_primary_effect(effect_sizes: dict) -> dict:
    """Get the primary/largest effect size from the dict."""
    if not effect_sizes:
        return {"effect_size": "N/A", "p_value": "N/A"}
    
    # Find the feature with the largest absolute effect size
    primary_key = max(effect_sizes.keys(), 
                       key=lambda k: abs(effect_sizes[k].get("effect_size", 0)))
    return effect_sizes[primary_key]
```

**Step 1 — Word budget allocation:**

```python
def _allocate_word_budget(rubric: dict) -> dict:
    """
    Allocate the writeup's word budget based on rubric weights.
    More points on a criterion → more words for the section that addresses it.
    """
    max_words = rubric.get("writeup_template", {}).get("max_words", 2000)
    criteria = rubric.get("criteria", [])
    template_sections = rubric.get("writeup_template", {}).get("sections", [])
    
    if not criteria or not template_sections:
        # Default equal allocation
        per_section = max_words // max(len(template_sections), 5)
        return {section: per_section for section in template_sections}
    
    # Map each criterion to the writeup section it scores
    CRITERION_TO_SECTION = {
        "clinical": "problem_statement",
        "domain": "problem_statement",
        "relevance": "problem_statement",
        "technical": "methodology",
        "method": "methodology",
        "code": "methodology",
        "model": "methodology",
        "document": "methodology",  # Methodology clarity
        "writeup": "results",       # Results clarity
        "clarity": "results",
        "insight": "results",
        "finding": "results",
        "result": "results",
        "analysis": "results",
        "novel": "discussion",
        "impact": "discussion",
        "creative": "discussion",
    }
    
    section_weights = {}
    for criterion in criteria:
        name_lower = criterion["name"].lower()
        matched_section = None
        for keyword, section in CRITERION_TO_SECTION.items():
            if keyword in name_lower:
                matched_section = section
                break
        
        if matched_section:
            section_weights[matched_section] = section_weights.get(matched_section, 0) + criterion.get("weight", 20)
        else:
            # Unmatched — distribute to results
            section_weights["results"] = section_weights.get("results", 0) + criterion.get("weight", 20)
    
    # Add fixed sections
    section_weights.setdefault("limitations", 10)  # Always need limitations
    section_weights.setdefault("reproducibility", 5)  # Always need repro notes
    
    # Normalize to word budget
    total_weight = sum(section_weights.values())
    word_budget = {}
    for section, weight in section_weights.items():
        words = max(100, int(max_words * weight / total_weight))
        word_budget[section] = words
    
    # Ensure total doesn't exceed max
    while sum(word_budget.values()) > max_words:
        largest = max(word_budget, key=word_budget.get)
        word_budget[largest] -= 10
    
    return word_budget
```

```python
    # === REPRODUCIBILITY ===
    sections["reproducibility"] = (
```

**Step 3 — Assemble and polish:**

```python
def _assemble_writeup(sections: dict, thesis: dict, rubric: dict) -> str:
    """Assemble sections into final markdown with headers."""
    
    # Generate a compelling title
    title = thesis.get("statement", "Analysis")
    if len(title) > 80:
        title = title[:77] + "..."
    
    # Map section keys to display headers (competition-specific)
    SECTION_HEADERS = {
        "problem_statement": "Clinical Problem Statement",
        "methodology": "Methodology",
        "results": "Results",
        "discussion": "Discussion & Clinical Implications",
        "limitations": "Limitations",
        "reproducibility": "Reproducibility Notes",
    }
    
    # Adapt headers to match the competition's template
    template_sections = rubric.get("writeup_template", {}).get("sections", [])
    for ts in template_sections:
        # Try to match template section names to our sections
        ts_lower = ts.lower().replace("_", " ")
        for key in SECTION_HEADERS:
            if key.replace("_", " ") in ts_lower or ts_lower in key.replace("_", " "):
                SECTION_HEADERS[key] = ts  # Use competition's exact naming
    
    writeup = f"# {title}\n\n"
    
    for section_key in ["problem_statement", "methodology", "results", "discussion", "limitations", "reproducibility"]:
        if section_key in sections:
            header = SECTION_HEADERS.get(section_key, section_key.replace("_", " ").title())
            writeup += f"## {header}\n\n{sections[section_key]}\n\n"
    
    return writeup


def _polish_writeup(writeup: str, rubric: dict, max_words: int, n_passes: int) -> str:
    """
    Run N polish passes on the writeup.
    Each pass targets a specific improvement based on rubric criteria.
    """
    from tools.llm_provider import llm_call
    
    criteria_block = "\n".join([
        f"  - {c['name']} ({c['weight']} pts): {c.get('top_score_description', c.get('description', ''))}"
        for c in rubric.get("criteria", [])
    ])
    
    for pass_num in range(n_passes):
        # Each pass has a different focus
        if pass_num == 0:
            focus = "Sharpen the problem statement. Make findings SPECIFIC with exact numbers. Remove generic filler."
        elif pass_num == 1:
            focus = "Ensure methodology is clearly justified. Every choice should have a reason. Strengthen the argument flow."
        else:
            focus = "Polish language. Ensure word count is under limit. Make the writeup compelling to a judge who reads 50 submissions."
        
        current_word_count = len(writeup.split())
        
        response = llm_call(
            prompt=f"""You are an expert editor for data science competition writeups.

RUBRIC (what judges score on):
{criteria_block}

CURRENT WRITEUP (pass {pass_num + 1} of {n_passes}, {current_word_count} words):
---
{writeup}
---

FOCUS FOR THIS PASS: {focus}

HARD CONSTRAINT: Final writeup must be under {max_words} words. Current: {current_word_count}.
{"TRIM AGGRESSIVELY — cut filler, merge redundant sentences." if current_word_count > max_words else ""}

Return the improved writeup ONLY. No commentary. No markdown fences around the output.""",
            agent_name="narrative_engine",
            temperature=0.3,
        )
        
        writeup = response["text"].strip()
    
    return writeup
```

**Step 4 — Main function:**

```python
def generate_hackathon_writeup(state: ProfessorState, session_dir: str) -> str:
    """
    Generate the complete hackathon writeup.
    Returns path to the writeup file.
    """
    rubric = state.hackathon_rubric or {}
    effort_plan = state.hackathon_effort_plan or {}
    max_words = rubric.get("writeup_template", {}).get("max_words", 2000)
    n_passes = effort_plan.get("narrative_polish_passes", 2)
    
    # Step 1: Allocate word budget
    word_budget = _allocate_word_budget(rubric)
    
    # Step 2: Generate sections
    sections = _generate_writeup_sections(state, session_dir, word_budget)
    
    # Step 3: Assemble
    writeup = _assemble_writeup(sections, state.active_thesis or {}, rubric)
    
    # Step 4: Polish
    writeup = _polish_writeup(writeup, rubric, max_words, n_passes)
    
    # Step 5: Final word count check
    final_count = len(writeup.split())
    if final_count > max_words:
        # Emergency trim — ask LLM to cut to exactly max_words
        from tools.llm_provider import llm_call
        writeup = llm_call(
            prompt=f"Cut this writeup to EXACTLY {max_words} words. Remove the least important sentences. "
                   f"Preserve all section headers and key statistics.\n\n{writeup}",
            agent_name="narrative_engine", temperature=0.2,
        )["text"]
    
    # Save
    writeup_path = os.path.join(session_dir, "hackathon_writeup.md")
    with open(writeup_path, "w") as f:
        f.write(writeup)
    
    return writeup_path
```

## WHAT NOT TO DO

- Do NOT integrate more than 3 external datasets. HARDCODED cap.
- Do NOT let joins change the competition data row count. Verify with Data Integrity.
- Do NOT construct reference tables without citing the source. Every table must reference a published guideline.
- Do NOT generate generic plot titles like "Figure 1: Distribution." Titles MUST state the finding.
- Do NOT let the writeup exceed the word limit. Emergency trim if needed.
- Do NOT skip the limitations section. Judges REWARD honesty — it scores points on Documentation.
- Do NOT use temperature > 0.4 for writeup generation. We want precision, not creativity in the writeup.
- Do NOT use Pandas for data loading. Polars only. Convert at the matplotlib boundary.
- Do NOT include code in the writeup. It goes in the notebook.
- Do NOT let any component crash if its input is missing. External data not found → proceed without it. Plots fail → log error, continue. Writeup section fails → insert "[Section generation failed]" and continue.

```python
def _build_hackathon_feature_prompt(
    state: ProfessorState,
    round_num: int,
    existing_feature_names: list[str],
) -> str:
    """
    Modified feature generation prompt for hackathon mode.
    
    Instead of "generate features that improve CV score," this asks
    "generate features that TEST the active thesis."
    
    The features fall into 4 categories that systematically build
    the evidence for or against the thesis:
    
    1. CONDITION features — identify when/where the thesis condition applies
    2. OUTCOME features — measure the outcome within each condition group
    3. DELTA features — quantify the difference between groups
    4. MODERATOR features — find what makes the effect stronger or weaker
    """
    thesis = state.active_thesis
    if not thesis:
        # Fallback to traditional prompt if no thesis
        return _build_feature_prompt(state, round_num, existing_feature_names)
    
    # Determine which data to use
    data_path = state.enriched_data_path or state.features_train_path or state.clean_data_path
    
    # External data context
    external_info = ""
    if state.external_datasets:
        external_info = "EXTERNAL DATA INTEGRATED:\n"
        for ds in state.external_datasets:
            external_info += f"  - {ds.get('name', '?')}: joined on {ds.get('join_key', '?')}, "
            external_info += f"{ds.get('integration_result', {}).get('new_columns', ['?'])} new columns\n"
    
    # Domain context for informed feature ideas
    domain_recipes = ""
    if state.domain_brief and state.domain_brief.get("feature_recipes"):
        recipes = state.domain_brief["feature_recipes"][:5]
        domain_recipes = "DOMAIN FEATURE RECIPES (from domain research):\n"
        for r in recipes:
            domain_recipes += f"  - {r.get('name', '?')}: {r.get('formula', '?')} — {r.get('rationale', '')[:100]}\n"
    
    # Round-specific direction
    round_directions = {
        1: "Focus on CONDITION features — create the binary/categorical splits that define your comparison groups.",
        2: "Focus on OUTCOME and DELTA features — compute the target-related metric within each group and the gap between groups.",
        3: "Focus on MODERATOR features — cross the condition with other variables to find what amplifies or dampens the effect.",
        4: "Focus on REFINEMENT — compute second-order interactions, subgroup-specific deltas, and edge-case features.",
        5: "Focus on ROBUSTNESS — alternative operationalizations of the condition, sensitivity checks, complementary metrics.",
    }
    round_direction = round_directions.get(round_num, round_directions[3])

    prompt = f"""You are a senior data scientist testing a specific thesis for a hackathon competition.

THESIS: "{thesis.get('statement', '')}"
HYPOTHESIS: "{thesis.get('hypothesis', '')}"
CONDITION VARIABLE: {thesis.get('condition_variable', '')}
TARGET AUDIENCE: {thesis.get('target_audience', '')}

ROUND {round_num} DIRECTION: {round_direction}

Generate features in these 4 categories:

1. CONDITION FEATURES — identify WHEN/WHERE the thesis condition applies
   Create binary or categorical variables that split the data into 
   "condition present" vs "condition absent" groups.
   The condition variable is: {thesis.get('condition_variable', '')}
   
   Examples for this thesis:
   - Binary flag: is_condition_present (1 if condition applies, 0 otherwise)
   - Categorical: condition_subgroup (fine-grained split of the condition)
   - Derived: condition_severity (continuous measure of how strongly the condition applies)

2. OUTCOME FEATURES — measure the outcome WITHIN each condition group
   Compute the target-related metric separately for each condition.
   These features capture what happens INSIDE each group.
   
   Examples:
   - outcome_when_condition_present (aggregate target within condition=1 group)
   - outcome_when_condition_absent (aggregate target within condition=0 group)
   - rate_metric_by_condition (rate or proportion per group)

3. DELTA FEATURES — quantify the DIFFERENCE between groups
   This is the feature that DIRECTLY measures the thesis.
   A large, statistically significant delta PROVES the thesis.
   
   Examples:
   - condition_gap = outcome_present - outcome_absent
   - condition_ratio = outcome_present / (outcome_absent + 1e-8)
   - relative_change = (outcome_present - outcome_absent) / (outcome_absent + 1e-8)

4. MODERATOR FEATURES — find what makes the effect STRONGER or WEAKER
   Cross the condition with other variables to discover moderators.
   
   Examples:
   - condition_gap × moderator_variable (interaction)
   - condition_gap_by_subgroup (delta computed within subgroups)
   - is_high_risk_subgroup (binary flag for subgroups where the effect is strongest)

{domain_recipes}

{chr(10).join(f'  - {name}' for name in existing_feature_names[:30])}
```

### Addition 2: Hackathon gate function

```python
def _hackathon_gate(
    feature_values: np.ndarray,
    condition_values: np.ndarray,
    gate_config: dict,
) -> tuple[bool, float, float, dict]:
    """
    Hackathon-mode statistical gate.
    
    Instead of testing "does this feature improve CV score?" (Wilcoxon on fold scores),
    tests "does this feature show a statistically significant conditional effect?"
    (Mann-Whitney U on group values).
    
    This is the right question for hackathons: we're not optimizing a model,
    we're PROVING a thesis. A feature that shows a significant conditional
    effect is evidence FOR the thesis, regardless of whether it improves CV.
    
    Args:
        feature_values: The feature values for all training rows (numpy array)
        condition_values: Binary array (0/1) indicating condition group membership
        gate_config: Adaptive thresholds from get_gate_config()
    
    Returns:
        (passed, p_value, effect_size, effect_details)
    """
    from scipy.stats import mannwhitneyu
    
    # Split by condition
    mask_a = condition_values == 1
    mask_b = condition_values == 0
    
    group_a = feature_values[mask_a]
    group_b = feature_values[mask_b]
    
    # Minimum group size check — need enough observations for a valid test
    MIN_GROUP_SIZE = 30
    if len(group_a) < MIN_GROUP_SIZE or len(group_b) < MIN_GROUP_SIZE:
        return False, 1.0, 0.0, {
            "reason": f"Insufficient group sizes: group_a={len(group_a)}, group_b={len(group_b)} (min={MIN_GROUP_SIZE})",
            "group_a_n": int(len(group_a)),
            "group_b_n": int(len(group_b)),
        }
    
    # Drop NaN from both groups
    group_a = group_a[~np.isnan(group_a)]
    group_b = group_b[~np.isnan(group_b)]
    
    if len(group_a) < MIN_GROUP_SIZE or len(group_b) < MIN_GROUP_SIZE:
        return False, 1.0, 0.0, {
            "reason": f"Insufficient non-null values: group_a={len(group_a)}, group_b={len(group_b)}",
            "group_a_n": int(len(group_a)),
            "group_b_n": int(len(group_b)),
        }
    
    # Mann-Whitney U test: do the groups differ significantly?
    # Two-sided — we don't know the direction a priori
    try:
        stat, p_value = mannwhitneyu(group_a, group_b, alternative="two-sided")
    except ValueError as e:
        # All values identical or other edge case
        return False, 1.0, 0.0, {"reason": f"Mann-Whitney failed: {str(e)}"}
    
    # Cohen's d effect size
    # Pooled standard deviation
    pooled_std = np.sqrt((group_a.std()**2 + group_b.std()**2) / 2)
    
    if pooled_std < 1e-10:
        # Zero variance — groups are constant
        effect_size = 0.0
    else:
        effect_size = (group_a.mean() - group_b.mean()) / pooled_std
    
    # Pass criteria: statistically significant AND meaningful effect size
    # Use the adaptive p-value threshold from gate_config
    p_threshold = gate_config.get("wilcoxon_p", 0.05)
    
    # Effect size thresholds (Cohen's conventions):
    # |d| > 0.2 = small effect (minimum for hackathon relevance)
    # |d| > 0.5 = medium effect (strong evidence)
    # |d| > 0.8 = large effect (very strong evidence)
    MINIMUM_EFFECT_SIZE = 0.2
    
    passed = (p_value < p_threshold) and (abs(effect_size) > MINIMUM_EFFECT_SIZE)
    
    # Build detailed effect report
    effect_details = {
        "p_value": float(p_value),
        "effect_size": float(effect_size),
        "effect_magnitude": (
            "large" if abs(effect_size) > 0.8 else
            "medium" if abs(effect_size) > 0.5 else
            "small" if abs(effect_size) > 0.2 else
            "negligible"
        ),
        "group_a_mean": float(group_a.mean()),
        "group_b_mean": float(group_b.mean()),
        "group_a_std": float(group_a.std()),
        "group_b_std": float(group_b.std()),
        "group_a_n": int(len(group_a)),
        "group_b_n": int(len(group_b)),
        "group_a_median": float(np.median(group_a)),
        "group_b_median": float(np.median(group_b)),
        "direction": "group_a > group_b" if group_a.mean() > group_b.mean() else "group_b > group_a",
        "mann_whitney_U": float(stat),
    }
    
    return passed, float(p_value), float(effect_size), effect_details
```

### How the condition values are determined

The hackathon gate needs `condition_values` — a binary array splitting the data into two groups. This comes from the CONDITION FEATURES generated in round 1.

```python
def _get_condition_values(
    df,  # Polars DataFrame
    thesis: dict,
    feature_manifest: list,
) -> np.ndarray:
    """
    Get the binary condition split for hackathon gate testing.
    
    Looks for condition features (generated in round 1) in the feature manifest.
    If no condition feature exists yet (round 1 is generating them), 
    tries to derive the condition from the thesis's condition_variable description.
    
    Returns:
        numpy array of 0s and 1s, same length as df
    """
    # Strategy 1: Find an existing condition feature in the manifest
    condition_features = [
        f for f in feature_manifest
        if f.get("category") == "condition" or 
        "is_" in f.get("name", "") or
        "_flag" in f.get("name", "") or
        f.get("name", "").startswith("condition_")
    ]
    
    if condition_features:
        # Use the first condition feature
        condition_col = condition_features[0]["name"]
        if condition_col in df.columns:
            values = df[condition_col].to_numpy()
            # Ensure binary
            if len(np.unique(values[~np.isnan(values)])) <= 2:
                return (values == values[~np.isnan(values)].max()).astype(int)
            else:
                # Multi-value condition — split at median
                median_val = np.nanmedian(values)
                return (values > median_val).astype(int)
    
    # Strategy 2: Derive from thesis condition_variable
    condition_var = thesis.get("condition_variable", "")
    
    # Try to find a column matching the condition variable name
    for col in df.columns:
        if col.lower() in condition_var.lower() or condition_var.lower() in col.lower():
            values = df[col].to_numpy()
            if df[col].dtype in [pl.Utf8, pl.Categorical]:
                # Categorical — use most common value as one group, rest as other
                most_common = df[col].mode()[0]
                return (df[col] == most_common).to_numpy().astype(int)
            else:
                # Numeric — split at median
                median_val = np.nanmedian(values)
                return (values > median_val).astype(int)
    
    # Strategy 3: Fallback — can't determine condition, use random split
    # This is a degradation mode — the gate will likely not find meaningful effects
    np.random.seed(42)
    return np.random.randint(0, 2, size=len(df))
```

### How the Feature Factory knows which features are "condition" vs "delta" vs "moderator"

The LLM is instructed to tag each feature with a CATEGORY comment in the generated code:


```python
def _extract_feature_categories(code: str, feature_names: list) -> dict:
    """
    Parse CATEGORY tags from the generated code to classify features.
    
    Returns: {feature_name: category}
    """
    categories = {}
    lines = code.split("\n")
    current_category = "uncategorized"
    
    for line in lines:
        stripped = line.strip()
        
        # Check for category comment
        if stripped.startswith("# CATEGORY:"):
            category_str = stripped.replace("# CATEGORY:", "").strip().lower()
            if category_str in ("condition", "outcome", "delta", "moderator"):
                current_category = category_str
        
        # Check for alias (feature name)
        if ".alias(" in stripped:
            import re
            alias_match = re.search(r'\.alias\(["\']([^"\']+)["\']\)', stripped)
            if alias_match:
                fname = alias_match.group(1)
                if fname in feature_names:
                    categories[fname] = current_category
    
    # Fallback: infer from naming conventions
    for fname in feature_names:
        if fname not in categories:
            fname_lower = fname.lower()
            if fname_lower.startswith("is_") or "_flag" in fname_lower or "condition" in fname_lower:
                categories[fname] = "condition"
            elif "gap" in fname_lower or "delta" in fname_lower or "diff" in fname_lower or "ratio" in fname_lower:
                categories[fname] = "delta"
            elif "moderator" in fname_lower or "interaction" in fname_lower or "_x_" in fname_lower:
                categories[fname] = "moderator"
            elif "rate" in fname_lower or "accuracy" in fname_lower or "outcome" in fname_lower:
                categories[fname] = "outcome"
            else:
                categories[fname] = "uncategorized"
    
    return categories
```

```python
class TestHackathonGate:
    
    def test_significant_effect_passes(self):
        """Feature with significant effect and |d| > 0.2 passes the gate."""
        np.random.seed(42)
        # Group A: mean=5, Group B: mean=3 — clear difference
        feature = np.concatenate([
            np.random.normal(5, 1, 500),  # condition=1

        ])
        condition = np.array([1]*500 + [0]*500)

        assert passed == True
        assert p_value < 0.05
        assert abs(effect_size) > 0.2
        assert details["effect_magnitude"] in ("medium", "large")

        """Feature with no real difference between groups fails the gate."""
        np.random.seed(42)
        # Both groups: same distribution
        feature = np.random.normal(5, 1, 1000)
        condition = np.array([1]*500 + [0]*500)
        
        passed, p_value, effect_size, details = _hackathon_gate(

        feature = np.concatenate([
            np.random.normal(5.01, 1, 5000),  # condition=1
            np.random.normal(5.00, 1, 5000),  # condition=0
        ])

        passed, p_value, effect_size, details = _hackathon_gate(
```
